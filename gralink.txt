    <script>
        /* GLOBAL STATE */
        let state = 'intro';
        let mainTimer;
        
        const show = (id) => {
            document.querySelectorAll('#wrapper > div').forEach(d => d.classList.add('hidden'));
            document.getElementById('wrapper').classList.remove('shake');
            document.getElementById(id).classList.remove('hidden');
            if(id.startsWith('screen-lvl')) document.getElementById('status').classList.remove('hidden');
            else document.getElementById('status').classList.add('hidden');
        };

        const updateBar = (p) => document.getElementById('bar').style.width = p + '%';
        const fail = () => {
            clearInterval(mainTimer);
            show('screen-fail');
        };

        /* LEVEL 1: MEMORY */
        let sequence = [];
        let playerSeq = [];
        let isShowing = false;

        function initLvl1() {
            show('screen-lvl1');
            document.getElementById('lvl-title').innerText = "LVL 1: LOGIC";
            sequence = Array.from({length: 5}, () => Math.floor(Math.random() * 4));
            playerSeq = [];
            setTimeout(playSequence, 1000);
        }

        async function playSequence() {
            isShowing = true;
            document.getElementById('mem-msg').innerText = "Watch...";
            for (let s of sequence) {
                await new Promise(r => setTimeout(r, 600));
                flash(s);
            }
            isShowing = false;
            document.getElementById('mem-msg').innerText = "Repeat the pattern!";
        }

        function flash(i) {
            const pad = document.getElementById(`pad-${i}`);
            pad.classList.add('active-pad');
            setTimeout(() => pad.classList.remove('active-pad'), 400);
        }

        function playerTap(i) {
            if (isShowing) return;
            flash(i);
            playerSeq.push(i);
            if (playerSeq[playerSeq.length - 1] !== sequence[playerSeq.length - 1]) return fail();
            updateBar((playerSeq.length / sequence.length) * 100);
            if (playerSeq.length === sequence.length) setTimeout(initLvl2, 800);
        }

        /* LEVEL 2: PHYSICS (Slingshot) */
        let isDragging = false;
        let pPos = { x: 50, y: 280 };
        let startPos = { x: 50, y: 280 };
        let velocity = { x: 0, y: 0 };
        let launched = false;

        function initLvl2() {
            show('screen-lvl2');
            updateBar(0);
            document.getElementById('lvl-title').innerText = "LVL 2: PHYSICS";
            setupSlingshot();
        }

        function setupSlingshot() {
            const touch = document.getElementById('touch-layer');
            const proj = document.getElementById('projectile');
            const line = document.getElementById('slingshot-line');
            const rect = document.getElementById('physics-container').getBoundingClientRect();

            const moveHandler = (e) => {
                if (!isDragging || launched) return;
                const clientX = e.touches ? e.touches[0].clientX : e.clientX;
                const clientY = e.touches ? e.touches[0].clientY : e.clientY;
                
                const dx = (clientX - rect.left) - startPos.x;
                const dy = (clientY - rect.top) - startPos.y;
                
                // Limited stretch
                const dist = Math.sqrt(dx*dx + dy*dy);
                const limit = 80;
                const scale = dist > limit ? limit/dist : 1;
                
                pPos.x = startPos.x + dx * scale;
                pPos.y = startPos.y + dy * scale;
                
                proj.setAttribute('cx', pPos.x);
                proj.setAttribute('cy', pPos.y);
                
                line.classList.remove('hidden');
                line.setAttribute('x2', pPos.x);
                line.setAttribute('y2', pPos.y);
                
                velocity.x = (startPos.x - pPos.x) * 0.15;
                velocity.y = (startPos.y - pPos.y) * 0.15;
            };

            touch.addEventListener('mousedown', () => isDragging = true);
            touch.addEventListener('touchstart', (e) => { e.preventDefault(); isDragging = true; });
            
            window.addEventListener('mousemove', moveHandler);
            window.addEventListener('touchmove', moveHandler);

            const release = () => {
                if (!isDragging || launched) return;
                isDragging = false;
                line.classList.add('hidden');
                runPhysics();
            };

            window.addEventListener('mouseup', release);
            window.addEventListener('touchend', release);
        }

        function runPhysics() {
            launched = true;
            const proj = document.getElementById('projectile');
            const target = document.getElementById('phys-target');
            const tx = parseFloat(target.getAttribute('cx')) / 100 * 320; // scale based on container size
            const ty = parseFloat(target.getAttribute('cy')) / 100 * 320;
            
            let curX = pPos.x;
            let curY = pPos.y;

            const loop = setInterval(() => {
                velocity.y += 0.2; // Gravity
                curX += velocity.x;
                curY += velocity.y;
                
                proj.setAttribute('cx', curX);
                proj.setAttribute('cy', curY);

                // Check hit (rough distance)
                const d = Math.sqrt(Math.pow(curX - 272, 2) + Math.pow(curY - 48, 2));
                if (d < 25) {
                    clearInterval(loop);
                    initLvl3();
                }

                // Check out of bounds
                if (curX > 400 || curY > 400 || curX < -50) {
                    clearInterval(loop);
                    resetPhys();
                }
            }, 20);
        }

        function resetPhys() {
            launched = false;
            pPos = { x: 50, y: 280 };
            const proj = document.getElementById('projectile');
            proj.setAttribute('cx', 50);
            proj.setAttribute('cy', 280);
            document.getElementById('physics-container').classList.add('shake');
            setTimeout(() => document.getElementById('physics-container').classList.remove('shake'), 200);
        }

        /* LEVEL 3: REFLEX */
        let molesHit = 0;
        let moleTimer;
        let gameTime = 10.0;

        function initLvl3() {
            show('screen-lvl3');
            updateBar(0);
            document.getElementById('lvl-title').innerText = "LVL 3: REFLEX";
            molesHit = 0;
            
            mainTimer = setInterval(() => {
                gameTime -= 0.1;
                document.getElementById('timer').innerText = gameTime.toFixed(1) + 's';
                if(gameTime <= 0) fail();
            }, 100);
            
            spawnMole();
        }

        function spawnMole() {
            const mole = document.getElementById('mole');
            const arena = document.getElementById('tap-arena');
            
            mole.classList.remove('hidden');
            mole.style.left = Math.random() * (arena.clientWidth - 50) + 'px';
            mole.style.top = Math.random() * (arena.clientHeight - 50) + 'px';
            
            // Mole stays for less time as you progress
            clearTimeout(moleTimer);
            moleTimer = setTimeout(() => {
                mole.classList.add('hidden');
                if(molesHit < 10) spawnMole();
            }, Math.max(400, 1000 - (molesHit * 80)));
        }

        function moleHit() {
            molesHit++;
            updateBar((molesHit / 10) * 100);
            document.getElementById('mole').classList.add('hidden');
            if (molesHit >= 10) {
                clearInterval(mainTimer);
                show('screen-win');
            } else {
                spawnMole();
            }
        }
